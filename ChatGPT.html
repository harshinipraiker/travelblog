Below is a simple C program demonstrating undo and redo functionality for text editing operations using a stack. This program allows the user to perform typing, deleting, and formatting actions. It maintains a stack to store the history of text edits, allowing for undo and redo operations.

c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_TEXT_SIZE 1000

// Node structure for stack
typedef struct Node {
    char text[MAX_TEXT_SIZE];
    struct Node* next;
} Node;

// Function prototypes
void push(Node** top, const char* text);
char* pop(Node** top);
void undo(Node** current, Node** undo_stack, Node** redo_stack);
void redo(Node** current, Node** undo_stack, Node** redo_stack);

int main() {
    Node* undo_stack = NULL;
    Node* redo_stack = NULL;
    Node* current = NULL;
    char input[MAX_TEXT_SIZE];
    char option;

    printf("Text Editor\n");

    do {
        printf("\nOptions:\n");
        printf("1. Type text\n");
        printf("2. Delete text\n");
        printf("3. Format text\n");
        printf("4. Undo\n");
        printf("5. Redo\n");
        printf("6. Exit\n");
        printf("Enter option: ");
        scanf(" %c", &option);

        switch (option) {
            case '1':
                printf("Enter text to type: ");
                scanf(" %[^\n]", input);
                push(&undo_stack, input);
                current = NULL;
                break;
            case '2':
                printf("Text deleted.\n");
                push(&undo_stack, "");
                current = NULL;
                break;
            case '3':
                printf("Text formatted.\n");
                push(&undo_stack, "Formatted text");
                current = NULL;
                break;
            case '4':
                undo(&current, &undo_stack, &redo_stack);
                break;
            case '5':
                redo(&current, &undo_stack, &redo_stack);
                break;
            case '6':
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid option\n");
        }
    } while (option != '6');

    return 0;
}

// Function to push a new text onto the stack
void push(Node** top, const char* text) {
    Node* newNode = (Node*)malloc(sizeof(Node));
    if (newNode == NULL) {
        printf("Memory allocation failed.\n");
        exit(EXIT_FAILURE);
    }
    strncpy(newNode->text, text, MAX_TEXT_SIZE - 1);
    newNode->text[MAX_TEXT_SIZE - 1] = '\0';
    newNode->next = *top;
    *top = newNode;
}

// Function to pop the top text from the stack
char* pop(Node** top) {
    if (*top == NULL) {
        printf("Stack is empty.\n");
        return NULL;
    }
    Node* temp = *top;
    char* text = temp->text;
    *top = temp->next;
    free(temp);
    return text;
}

// Function to perform undo operation
void undo(Node** current, Node** undo_stack, Node** redo_stack) {
    if (*undo_stack == NULL) {
        printf("Nothing to undo.\n");
        return;
    }
    push(redo_stack, pop(undo_stack));
    *current = NULL;
    printf("Undo successful.\n");
}

// Function to perform redo operation
void redo(Node** current, Node** undo_stack, Node** redo_stack) {
    if (*redo_stack == NULL) {
        printf("Nothing to redo.\n");
        return;
    }
    push(undo_stack, pop(redo_stack));
    *current = NULL;
    printf("Redo successful.\n");
}

This program provides a simple command-line interface for text editing. When the user chooses options like typing text, deleting text, or formatting text, the respective operations are pushed onto the undo stack. Undoing an action pops it from the undo stack and pushes it onto the redo stack. Redoing an action pops it from the redo stack and pushes it back onto the undo stack.

    // Free memory allocated for the playlist

// Function to create a new song node
Song* createSong(const char* title, const char* artist, int duration) {
    Song* newSong = (Song*)malloc(sizeof(Song));
    if (newSong != NULL) {
        strcpy(newSong->title, title);
        strcpy(newSong->artist, artist);
        newSong->duration = duration;
        newSong->next = NULL;
    }
    return newSong;
}

// Function to add a song to the playlist
void addSong(Song** playlist, const char* title, const char* artist, int duration) {
    Song* newSong = createSong(title, artist, duration);
    if (newSong == NULL) {
        printf("Failed to add song. Memory allocation error.\n");
        return;
    }

    if (*playlist == NULL) {
        *playlist = newSong;
    } else {
        Song* current = *playlist;
        while (current->next != NULL) {
            current = current->next;
        }
        current->next = newSong;
    }
}

// Function to display the playlist
void displayPlaylist(Song* playlist) {
    printf("Playlist:\n");
    int songNumber = 1;
    while (playlist != NULL) {
        printf("%d. %s - %s (%d seconds)\n", songNumber, playlist->title, playlist->artist, playlist->duration);
        playlist = playlist->next;
        songNumber++;
    }
}
